# ThinCloud Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a self-hosted Chrome-streaming thin-client service that runs on a Mac Mini M4 and streams to any browser tab with zero client install.

**Architecture:** FFmpeg captures the Mac Mini display via AVFoundation and encodes H.264 with VideoToolbox hardware acceleration, outputting RTP to a local UDP socket. Node.js reads the RTP stream and injects it into a WebRTC peer connection (via `node-datachannel` or `wrtc`), which delivers the stream to the client browser. A WebSocket channel on the same Node.js server handles signaling and relays mouse/keyboard input back to Chrome via nut.js.

**Tech Stack:** Node.js, Express, ws, node-datachannel (primary) / wrtc (fallback), FFmpeg (AVFoundation + VideoToolbox), @nut-tree/nut-js, jsonwebtoken, mkcert, launchd, Vanilla JS client

## Critical Pre-Implementation: WebRTC Library Decision Gate

Before writing any pipeline code, the Mac Mini M4 must be checked for prebuilt WebRTC library binaries. Run this first deploy step manually once:

```bash
# On Mac Mini — test wrtc prebuilt binary
cd ~/thincloud
npm install wrtc
node -e "require('wrtc'); console.log('wrtc OK')" 2>&1
```

- If output is `wrtc OK` with no compile output → set `THINCLOUD_WEBRTC_LIB=wrtc` in `.env`
- If it starts compiling from source (you'll see `gyp` output) → abort, run `npm uninstall wrtc`, then:

```bash
npm install node-datachannel
node -e "require('node-datachannel'); console.log('node-datachannel OK')" 2>&1
```

Set `THINCLOUD_WEBRTC_LIB=node-datachannel` in `.env`.

**The two libraries use different pipeline modes in `capture.js` and `webrtc.js`:**
- `node-datachannel`: FFmpeg → H.264 RTP → UDP socket → Node.js → RTP injection
- `wrtc`: FFmpeg → raw I420 frames → stdout pipe → Node.js → RTCVideoSource.onFrame()

`check-webrtc.sh` (Task 9) automates this detection.

```js
const jwt = require('jsonwebtoken');

function issueToken() {
  const secret = process.env.THINCLOUD_SECRET;
  if (!secret) throw new Error('THINCLOUD_SECRET not set');
  return jwt.sign({ iss: 'thincloud' }, secret, { expiresIn: '8h' });
}

function verifyToken(token) {
  const secret = process.env.THINCLOUD_SECRET;
  if (!secret) throw new Error('THINCLOUD_SECRET not set');
  return jwt.verify(token, secret);
}

module.exports = { issueToken, verifyToken };
```

```js
require('dotenv').config();
const { spawn } = require('child_process');
const dgram = require('dgram');
const { EventEmitter } = require('events');

const RTP_PORT = 5004;
const WIDTH = 1920;
const HEIGHT = 1080;
const FPS = 30;
const BITRATE = '4000k';

function createCapture() {
  const emitter = new EventEmitter();
  let ffmpegProc = null;
  let udpSocket = null;

  function start() {
    const mode = process.env.THINCLOUD_WEBRTC_LIB || 'node-datachannel';

    if (mode === 'node-datachannel') {
      _startRTPMode();
    } else {
      _startRawMode();
    }
  }

  function _startRTPMode() {
    ffmpegProc = spawn('ffmpeg', [
      '-f', 'avfoundation',
      '-capture_cursor', '1',
      '-framerate', String(FPS),
      '-video_size', `${WIDTH}x${HEIGHT}`,
      '-i', '1',
      '-vcodec', 'h264_videotoolbox',
      '-b:v', BITRATE,
      '-r', String(FPS),
      '-f', 'rtp',
      `rtp://127.0.0.1:${RTP_PORT}`,
    ], { stdio: ['ignore', 'ignore', 'pipe'] });

    ffmpegProc.stderr.on('data', () => {}); // suppress output
    ffmpegProc.on('error', (err) => emitter.emit('error', err));
    ffmpegProc.on('exit', (code) => {
      if (code !== 0 && code !== null) emitter.emit('error', new Error(`FFmpeg exited ${code}`));
    });

    udpSocket = dgram.createSocket('udp4');
    udpSocket.bind(RTP_PORT, () => {
      udpSocket.on('message', (msg) => emitter.emit('data', msg));
    });
  }

  function _startRawMode() {
    const frameSize = WIDTH * HEIGHT * 3 / 2; // I420
    ffmpegProc = spawn('ffmpeg', [
      '-f', 'avfoundation',
      '-capture_cursor', '1',
      '-framerate', String(FPS),
      '-video_size', `${WIDTH}x${HEIGHT}`,
      '-i', '1',
      '-f', 'rawvideo',
      '-pix_fmt', 'yuv420p',
      '-r', String(FPS),
      'pipe:1',
    ], { stdio: ['ignore', 'pipe', 'pipe'] });

    let buf = Buffer.alloc(0);
    ffmpegProc.stdout.on('data', (chunk) => {
      buf = Buffer.concat([buf, chunk]);
      while (buf.length >= frameSize) {
        const frame = buf.slice(0, frameSize);
        buf = buf.slice(frameSize);
        emitter.emit('frame', {
          width: WIDTH,
          height: HEIGHT,
          data: new Uint8Array(frame),
        });
      }
    });

    ffmpegProc.on('error', (err) => emitter.emit('error', err));
    ffmpegProc.on('exit', (code) => {
      if (code !== 0 && code !== null) emitter.emit('error', new Error(`FFmpeg exited ${code}`));
    });
  }

  function stop() {
    if (ffmpegProc) {
      ffmpegProc.kill('SIGTERM');
      ffmpegProc = null;
    }
    if (udpSocket) {
      udpSocket.close();
      udpSocket = null;
    }
  }

  return Object.assign(emitter, { start, stop });
}

module.exports = { createCapture };
```

```js
require('dotenv').config();

/**
 * createPeer — creates a WebRTC peer that streams video to a remote client.
 *
 * @param {string} offerSdp — SDP offer string from client
 * @param {function} onIceCandidate — called with each local ICE candidate string
 * @returns {Promise<{ answerSdp: string, feedData: function, feedFrame: function, close: function }>}
 *
 * feedData(buf: Buffer) — for node-datachannel mode: inject RTP packet
 * feedFrame(frame: {width, height, data: Uint8Array}) — for wrtc mode: inject raw I420 frame
 */
async function createPeer(offerSdp, onIceCandidate = () => {}) {
  const lib = process.env.THINCLOUD_WEBRTC_LIB;
  if (!lib) throw new Error('THINCLOUD_WEBRTC_LIB not set');

  if (lib === 'node-datachannel') {
    return _createNodeDatachannelPeer(offerSdp, onIceCandidate);
  } else if (lib === 'wrtc') {
    return _createWrtcPeer(offerSdp, onIceCandidate);
  } else {
    throw new Error(`Unknown THINCLOUD_WEBRTC_LIB: ${lib}`);
  }
}

async function _createNodeDatachannelPeer(offerSdp, onIceCandidate) {
  const dc = require('node-datachannel');

  const peer = new dc.PeerConnection('thincloud-server', {
    iceServers: [],
  });

  let videoTrack = null;

  return new Promise((resolve, reject) => {
    let answerSdp = null;

    peer.onLocalDescription((sdp, type) => {
      if (type === 'answer') answerSdp = sdp;
    });

    peer.onLocalCandidate((candidate, mid) => {
      onIceCandidate({ candidate, mid });
    });

    peer.onGatheringStateChange((state) => {
      if (state === 'complete' && answerSdp) {
        resolve({
          answerSdp,
          feedData: (buf) => {
            if (videoTrack) videoTrack.sendMessage(buf);
          },
          feedFrame: () => {},
          close: () => peer.close(),
        });
      }
    });

    try {
      peer.setRemoteDescription(offerSdp, 'offer');
      videoTrack = peer.addTrack('video');
      peer.setLocalDescription();
    } catch (err) {
      reject(err);
    }
  });
}

async function _createWrtcPeer(offerSdp, onIceCandidate) {
  const { RTCPeerConnection, RTCSessionDescription, nonstandard: { RTCVideoSource } } = require('wrtc');

  const source = new RTCVideoSource();
  const track = source.createTrack();
  const peer = new RTCPeerConnection({ iceServers: [] });
  peer.addTrack(track);

  peer.onicecandidate = ({ candidate }) => {
    if (candidate) onIceCandidate(candidate);
  };

  await peer.setRemoteDescription(new RTCSessionDescription({ type: 'offer', sdp: offerSdp }));
  const answer = await peer.createAnswer();
  await peer.setLocalDescription(answer);

  return {
    answerSdp: peer.localDescription.sdp,
    feedData: () => {},
    feedFrame: ({ width, height, data }) => {
      source.onFrame({ width, height, data });
    },
    close: () => peer.close(),
  };
}

module.exports = { createPeer };
```

```js
const { mouse, keyboard, Key, Point } = require('@nut-tree-fork/nut-js');

const BUTTON_MAP = {
  left: () => mouse.leftClick(),
  right: () => mouse.rightClick(),
};

const MODIFIER_MAP = {
  ctrl: Key.LeftControl,
  shift: Key.LeftShift,
  alt: Key.LeftAlt,
  meta: Key.LeftSuper,
};

async function handleInput(event) {
  switch (event.type) {
    case 'mousemove':
      await mouse.move([new Point(event.x, event.y)]);
      break;

    case 'mousedown': {
      const clickFn = BUTTON_MAP[event.button];
      if (clickFn) await clickFn();
      break;
    }

    case 'mouseup':
      // nut.js handles press+release in leftClick/rightClick — no separate mouseup needed
      break;

    case 'wheel':
      if (event.deltaY > 0) await mouse.scrollDown(Math.abs(event.deltaY));
      else if (event.deltaY < 0) await mouse.scrollUp(Math.abs(event.deltaY));
      break;

    case 'keydown': {
      const modKeys = (event.modifiers || []).map((m) => MODIFIER_MAP[m]).filter(Boolean);
      for (const k of modKeys) await keyboard.pressKey(k);
      if (event.key && event.key.length === 1) await keyboard.type(event.key);
      break;
    }

    case 'keyup': {
      const modKeys = (event.modifiers || []).map((m) => MODIFIER_MAP[m]).filter(Boolean);
      for (const k of modKeys) await keyboard.releaseKey(k);
      break;
    }

    default:
      break;
  }
}

module.exports = { handleInput };
```

```js
const path = require('path');
const os = require('os');

const CHROME_PATH = '/Applications/Google Chrome.app/Contents/MacOS/Google Chrome';

let chromeProc = null;

function launchChrome() {
  if (chromeProc) return; // already running

  chromeProc = spawn(CHROME_PATH, [
    `--user-data-dir=${PROFILE_DIR}`,
    '--no-first-run',
    '--disable-session-crashed-bubble',
    '--restore-last-session',

    detached: false,

  });
}

function killChrome() {
  if (chromeProc) {
    chromeProc.kill('SIGTERM');
    chromeProc = null;
  }
}

function isRunning() {
  return chromeProc !== null;
}

module.exports = { launchChrome, killChrome, isRunning };
```

Signaling state machine:
1. Client `POST /auth {secret}` → JWT
2. Client WebSocket connect `/signal?token=JWT` → verified on upgrade
3. Client sends `{ type: 'offer', sdp: '...' }` 
4. Server creates WebRTC peer, returns `{ type: 'answer', sdp: '...' }`
5. ICE candidates exchanged: `{ type: 'ice', candidate: '...' }`
6. On ICE connected: `capture.start()` + pipe data to peer
7. Client sends input events as `{ type: 'input', event: {...} }`
8. On WebSocket close: `capture.stop()`, `peer.close()`

```js
require('dotenv').config();
const fs = require('fs');
const https = require('https');
const path = require('path');
const express = require('express');
const { WebSocketServer } = require('ws');
const url = require('url');

const { issueToken, verifyToken } = require('./auth');
const { createCapture } = require('./capture');
const { createPeer } = require('./webrtc');
const { handleInput } = require('./input');
const { launchChrome } = require('./chrome');

function createApp() {
  const app = express();
  app.use(express.json());

  app.get('/', (req, res) => {
    res.sendFile(path.join(__dirname, 'client', 'index.html'));
  });

  app.post('/auth', (req, res) => {
    try {
      const provided = req.body && req.body.secret;
      const expected = process.env.THINCLOUD_SECRET;
      if (!provided || provided !== expected) {
        return res.status(401).json({ error: 'Unauthorized' });
      }
      const token = issueToken();
      res.json({ token });
    } catch (err) {
      res.status(401).json({ error: 'Unauthorized' });
    }
  });

  return app;
}

function startServer() {
  const app = createApp();
  const port = parseInt(process.env.PORT || '3000', 10);

  const certPath = process.env.TLS_CERT;
  const keyPath = process.env.TLS_KEY;

  let server;
  if (certPath && keyPath && fs.existsSync(certPath) && fs.existsSync(keyPath)) {
    server = https.createServer({
      cert: fs.readFileSync(certPath),
      key: fs.readFileSync(keyPath),
    }, app);
    console.log('[thincloud] HTTPS enabled');
  } else {
    const http = require('http');
    server = http.createServer(app);
    console.warn('[thincloud] WARNING: TLS cert not found, running HTTP (LAN only)');
  }

  // Single-client state
  let activeWs = null;
  let activePeer = null;
  let activeCapture = null;

  const wss = new WebSocketServer({ noServer: true });

  server.on('upgrade', (req, socket, head) => {
    const { pathname, query } = url.parse(req.url, true);
    if (pathname !== '/signal') {
      socket.destroy();
      return;
    }

    // Reject second client
    if (activeWs) {
      socket.write('HTTP/1.1 409 Conflict\r\n\r\n');
      socket.destroy();
      return;
    }

    const token = query.token;
    try {
      verifyToken(token);
    } catch {
      socket.write('HTTP/1.1 401 Unauthorized\r\n\r\n');
      socket.destroy();
      return;
    }

    wss.handleUpgrade(req, socket, head, (ws) => {
      wss.emit('connection', ws, req);
    });
  });

  wss.on('connection', (ws) => {
    activeWs = ws;
    launchChrome();
    console.log('[thincloud] Client connected');

    ws.on('message', async (raw) => {
      let msg;
      try { msg = JSON.parse(raw); } catch { return; }

      if (msg.type === 'offer') {
        try {
          activePeer = await createPeer(msg.sdp, (candidate) => {
            ws.send(JSON.stringify({ type: 'ice', candidate }));
          });
          ws.send(JSON.stringify({ type: 'answer', sdp: activePeer.answerSdp }));

          activeCapture = createCapture();
          activeCapture.on('data', (buf) => activePeer.feedData(buf));
          activeCapture.on('frame', (frame) => activePeer.feedFrame(frame));
          activeCapture.on('error', (err) => {
            console.error('[capture] error:', err.message);
            ws.close();
          });
          activeCapture.start();
        } catch (err) {
          console.error('[signaling] offer failed:', err.message);
          ws.close();
        }
      }

      if (msg.type === 'ice' && activePeer) {
        // ICE candidate from client — handled by WebRTC library internally
      }

      if (msg.type === 'input') {
        handleInput(msg.event).catch(() => {});
      }
    });

    ws.on('close', () => {
      console.log('[thincloud] Client disconnected');
      if (activeCapture) { activeCapture.stop(); activeCapture = null; }
      if (activePeer) { activePeer.close(); activePeer = null; }
      activeWs = null;
    });
  });

  server.listen(port, () => {
    console.log(`[thincloud] Listening on port ${port}`);
  });

  return server;
}

module.exports = { createApp, startServer };

if (require.main === module) {
  startServer();
}
```

- [ ] **Step 1: Create deploy.sh**

Create `scripts/deploy.sh`:

```bash
#!/usr/bin/env bash
set -e

# Configure these for your Mac Mini
MAC_MINI_HOST="${THINCLOUD_HOST:-mac-mini.local}"
MAC_MINI_USER="${THINCLOUD_USER:-$(whoami)}"
DEPLOY_PATH="/Users/Shared/thincloud"

echo "[deploy] Syncing to ${MAC_MINI_USER}@${MAC_MINI_HOST}:${DEPLOY_PATH}"

rsync -avz --progress \
  --exclude='.git' \
  --exclude='node_modules' \
  --exclude='.env' \
  --exclude='certs' \
  --exclude='logs' \
  "$(pwd)/" \
  "${MAC_MINI_USER}@${MAC_MINI_HOST}:${DEPLOY_PATH}/"

echo "[deploy] Installing dependencies..."
ssh "${MAC_MINI_USER}@${MAC_MINI_HOST}" "cd ${DEPLOY_PATH} && npm install --omit=dev"

echo "[deploy] Restarting service..."
ssh "${MAC_MINI_USER}@${MAC_MINI_HOST}" \
  "launchctl unload ~/Library/LaunchAgents/com.thincloud.plist 2>/dev/null || true; \
   launchctl load ~/Library/LaunchAgents/com.thincloud.plist"

echo "[deploy] Done. Service restarted."
```
